# Simple HTTP configuration 
server {
    listen 80;
    listen [::]:80;
    
    # Replace with your domain
    server_name mahd.shop www.mahd.shop;
    
    # Logging
    access_log /var/log/nginx/upstream-forwarder-access.log;
    error_log /var/log/nginx/upstream-forwarder-error.log;
    
    # Proxy all requests to Upstream forwarder
    location / {
        proxy_pass http://127.0.0.1:8787;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
